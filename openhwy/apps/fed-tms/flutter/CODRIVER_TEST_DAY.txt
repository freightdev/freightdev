═══════════════════════════════════════════════════════════════════════════════
                         WHAT TO WATCH FOR
═══════════════════════════════════════════════════════════════════════════════

✅ GOOD SIGNS:
  - All health checks return 200 OK
  - PIDs exist in ~/.codriver.d/var/runtime/pids/
  - Logs show "started successfully" or similar
  - Commands get routed correctly
  - Agents respond within 1-2 seconds
  - Database writes data

❌ BAD SIGNS:
  - Port already in use (kill old processes)
  - Connection refused (service didn't start)
  - Timeout (service crashed)
  - "Address already in use" (port conflict)
  - "No such file or directory" (path issue)
  - "Permission denied" (file permissions)

🤔 WEIRD SIGNS:
  - Service starts but doesn't respond (check firewall)
  - Logs show errors but service "works" (fix the errors)
  - Slow responses (>5 seconds) (performance issue)

                         TESTING SCRIPT (RUN THIS)
═══════════════════════════════════════════════════════════════════════════════

Save this as test-codriver.sh and run it:

#!/bin/bash
set -e

echo "🚛 CODRIVER TEST DAY"
echo ""

# 1. Clean slate
echo "1. Cleaning stale PIDs..."
find ~/.codriver.d/var/runtime/pids -name "*.pid" 2>/dev/null | while read pidfile; do
    pid=$(cat "$pidfile" 2>/dev/null)
    if ! ps -p $pid &> /dev/null 2>&1; then
        rm "$pidfile"
        echo "  Removed stale: $(basename $pidfile)"
    fi
done

# 2. Start SurrealDB
echo ""
echo "2. Starting SurrealDB..."
surreal start --log trace --user root --pass root \
  file:~/.codriver.d/var/data/surrealdb > ~/.codriver.d/var/logs/surrealdb.log 2>&1 &
sleep 3
if curl -s http://localhost:8000/health > /dev/null; then
    echo "  ✅ SurrealDB is up"
else
    echo "  ❌ SurrealDB failed"
    exit 1
fi

# 3. Start Coordinator
echo ""
echo "3. Starting Coordinator..."
cd ~/WORKSPACE/.ranger/src/coordinator
./target/release/coordinator > ~/.codriver.d/var/logs/coordinator.log 2>&1 &
echo $! > ~/.codriver.d/var/runtime/pids/coordinator.pid
sleep 3
if curl -s http://localhost:9999/health > /dev/null; then
    echo "  ✅ Coordinator is up"
else
    echo "  ❌ Coordinator failed"
    exit 1
fi

# 4. Start Microservices
echo ""
echo "4. Starting Microservices..."

# Auth
cd ~/.codriver.d/srv/auth.todo
go run cmd/server/main.go > ~/.codriver.d/var/logs/auth.log 2>&1 &
echo $! > ~/.codriver.d/var/runtime/pids/auth.pid
sleep 2
if curl -s http://localhost:8081/health > /dev/null; then
    echo "  ✅ Auth service is up"
else
    echo "  ⚠️  Auth service failed (continuing...)"
fi

# Email
cd ~/.codriver.d/srv/email.todo
go run cmd/server/main.go > ~/.codriver.d/var/logs/email.log 2>&1 &
echo $! > ~/.codriver.d/var/runtime/pids/email.pid
sleep 2
if curl -s http://localhost:8082/health > /dev/null; then
    echo "  ✅ Email service is up"
else
    echo "  ⚠️  Email service failed (continuing...)"
fi

# 5. Test end-to-end
echo ""
echo "5. Testing end-to-end command..."
response=$(curl -s -X POST http://localhost:9999/command \
  -H "Content-Type: application/json" \
  -d '{"command":"status","user_id":"test"}')

if [ ! -z "$response" ]; then
    echo "  ✅ Command succeeded"
    echo "  Response: $response"
else
    echo "  ❌ Command failed"
fi

echo ""
echo "🚛 Test complete! Check logs in ~/.codriver.d/var/logs/"


═══════════════════════════════════════════════════════════════════════════════
                         STOP SCRIPT (WHEN DONE TESTING)
═══════════════════════════════════════════════════════════════════════════════

Save this as stop-codriver.sh:

#!/bin/bash

echo "🛑 Stopping Codriver..."

# Kill all PIDs
find ~/.codriver.d/var/runtime/pids -name "*.pid" 2>/dev/null | while read pidfile; do
    pid=$(cat "$pidfile" 2>/dev/null)
    name=$(basename "$pidfile" .pid)
    if ps -p $pid &> /dev/null; then
        kill $pid
        echo "  Stopped: $name (PID $pid)"
    fi
    rm "$pidfile"
done

# Kill SurrealDB (might not have PID file)
pkill -f "surreal start" 2>/dev/null && echo "  Stopped: SurrealDB"

echo "🛑 All services stopped"
