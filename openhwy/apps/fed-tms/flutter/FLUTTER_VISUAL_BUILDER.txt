                         ARCHITECTURE LAYERS
═══════════════════════════════════════════════════════════════════════════════

Layer 1: CANVAS (Flutter)
  - Infinite scroll workspace
  - Drag-and-drop components
  - Connection lines (routes)
  - Zoom/pan controls
  - Grid snapping

Layer 2: COMPONENT LIBRARY (Flutter)
  - Atomic design structure:
    atoms/ → basic elements (buttons, inputs, text)
    molecules/ → simple composites (search bar, card header)
    organisms/ → complex composites (navbar, sidebar, full card)
    templates/ → page layouts
    pages/ → full screens
  
  - Each component:
    * Accepts props (size, color, text, etc.)
    * Has variants (primary, secondary, disabled, etc.)
    * Documented in Storybook-style viewer
    * Self-contained (no external dependencies)

Layer 3: ROUTING ENGINE (Rust)
  - Interprets visual connections
  - Generates Rust route handlers
  - Compiles routing code
  - Hot-reloads on changes

Layer 4: AI ASSISTANT (Ollama + Rust)
  - Reads README.md in each "spot" (component/route)
  - Understands context
  - Generates routing logic
  - Suggests connections
  - Auto-completes code

Layer 5: CODE GENERATOR (Rust)
  - Converts canvas → Flutter code
  - Converts connections → Rust routes
  - Generates type-safe interfaces
  - Outputs production-ready code

Layer 6: PREVIEW ENGINE (Flutter)
  - Live preview of app
  - Hot reload on changes
  - Responsive testing (mobile, tablet, desktop)
  - Interactive testing

```dart
import 'package:flutter/material.dart';

/// A reusable button atom with multiple variants
class AtomButton extends StatelessWidget {
  // Props (configurable)
  final String text;
  final VoidCallback? onPressed;
  final ButtonVariant variant;
  final ButtonSize size;
  final bool isLoading;
  final IconData? icon;
  final Color? customColor;
  
  const AtomButton({
    Key? key,
    required this.text,
    this.onPressed,
    this.variant = ButtonVariant.primary,
    this.size = ButtonSize.medium,
    this.isLoading = false,
    this.icon,
    this.customColor,
  }) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    return ElevatedButton(
      onPressed: isLoading ? null : onPressed,
      style: _getButtonStyle(),
      child: isLoading
          ? _buildLoadingIndicator()
          : _buildButtonContent(),
    );
  }
  
  ButtonStyle _getButtonStyle() {
    // Variant-specific styling
    switch (variant) {
      case ButtonVariant.primary:
        return ElevatedButton.styleFrom(
          backgroundColor: customColor ?? Colors.blue,
          foregroundColor: Colors.white,
          padding: _getPadding(),
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(8),
          ),
        );
      case ButtonVariant.secondary:
        return ElevatedButton.styleFrom(
          backgroundColor: Colors.grey[200],
          foregroundColor: Colors.black87,
          padding: _getPadding(),
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(8),
          ),
        );
      case ButtonVariant.outlined:
        return OutlinedButton.styleFrom(
          foregroundColor: customColor ?? Colors.blue,
          padding: _getPadding(),
          side: BorderSide(color: customColor ?? Colors.blue),
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.circular(8),
          ),
        );
      case ButtonVariant.text:
        return TextButton.styleFrom(
          foregroundColor: customColor ?? Colors.blue,
          padding: _getPadding(),
        );
    }
  }
  
  EdgeInsets _getPadding() {
    switch (size) {
      case ButtonSize.small:
        return EdgeInsets.symmetric(horizontal: 12, vertical: 8);
      case ButtonSize.medium:
        return EdgeInsets.symmetric(horizontal: 16, vertical: 12);
      case ButtonSize.large:
        return EdgeInsets.symmetric(horizontal: 24, vertical: 16);
    }
  }
  
  Widget _buildButtonContent() {
    if (icon == null) {
      return Text(text);
    }
    return Row(
      mainAxisSize: MainAxisSize.min,
      children: [
        Icon(icon, size: _getIconSize()),
        SizedBox(width: 8),
        Text(text),
      ],
    );
  }
  
  double _getIconSize() {
    switch (size) {
      case ButtonSize.small:
        return 16;
      case ButtonSize.medium:
        return 20;
      case ButtonSize.large:
        return 24;
    }
  }
  
  Widget _buildLoadingIndicator() {
    return SizedBox(
      width: _getIconSize(),
      height: _getIconSize(),
      child: CircularProgressIndicator(
        strokeWidth: 2,
        valueColor: AlwaysStoppedAnimation<Color>(Colors.white),
      ),
    );
  }
}

enum ButtonVariant {
  primary,
  secondary,
  outlined,
  text,
}

enum ButtonSize {
  small,
  medium,
  large,
}
```

A flexible, reusable button component.

| Prop | Type | Default | Description |

| onPressed | VoidCallback? | null | Callback when pressed |

                         CANVAS IMPLEMENTATION
═══════════════════════════════════════════════════════════════════════════════

The canvas is an infinite workspace where you drag components and connect them.

Key features:
  - Zoom in/out (pinch or mouse wheel)
  - Pan (drag background)
  - Grid snapping (optional)
  - Multi-select (Ctrl+click)
  - Delete (Del key)
  - Copy/paste (Ctrl+C/V)
  - Undo/redo (Ctrl+Z/Y)

Implementation: lib/canvas/canvas_widget.dart

```dart
import 'package:flutter/material.dart';
import 'canvas_controller.dart';
import 'canvas_node.dart';
import 'canvas_connection.dart';

class CanvasWidget extends StatefulWidget {
  final CanvasController controller;
  
  const CanvasWidget({
    Key? key,
    required this.controller,
  }) : super(key: key);
  
  @override
  _CanvasWidgetState createState() => _CanvasWidgetState();
}

class _CanvasWidgetState extends State<CanvasWidget> {
  double _scale = 1.0;
  Offset _offset = Offset.zero;
  Offset? _dragStart;
  
  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onScaleStart: _handleScaleStart,
      onScaleUpdate: _handleScaleUpdate,
      onScaleEnd: _handleScaleEnd,
      child: CustomPaint(
        painter: CanvasPainter(
          nodes: widget.controller.nodes,
          connections: widget.controller.connections,
          scale: _scale,
          offset: _offset,
        ),
        child: Stack(
          children: [
            // Grid background
            _buildGrid(),
            
            // Nodes (components)
            ...widget.controller.nodes.map((node) => _buildNode(node)),
            
            // Connections (lines)
            CustomPaint(
              painter: ConnectionPainter(
                connections: widget.controller.connections,
                scale: _scale,
                offset: _offset,
              ),
            ),
          ],
        ),
      ),
    );
  }
  
  Widget _buildGrid() {
    return CustomPaint(
      painter: GridPainter(
        scale: _scale,
        offset: _offset,
      ),
      child: Container(),
    );
  }
  
  Widget _buildNode(CanvasNode node) {
    return Positioned(
      left: node.position.dx * _scale + _offset.dx,
      top: node.position.dy * _scale + _offset.dy,
      child: Transform.scale(
        scale: _scale,
        child: GestureDetector(
          onPanUpdate: (details) => _handleNodeDrag(node, details),
          onTap: () => _handleNodeTap(node),
          child: node.component, // The actual Flutter widget
        ),
      ),
    );
  }
  
  // ... pan/zoom handlers
}
```

    _nodes.add(node);

═══════════════════════════════════════════════════════════════════════════════
                         AI-ASSISTED ROUTING
═══════════════════════════════════════════════════════════════════════════════

When you connect two components visually, the AI:

1. Reads README.md from both components
2. Understands what data flows between them
3. Generates Rust routing code
4. Validates the route
5. Hot-reloads the app

Example flow:

Canvas: [Button "Submit"] --connection--> [API "Create Job"]

Step 1: Read context
  From: components/atoms/button/README.md
    "Button emits onClick event with no data"
  
  To: components/organisms/api_card/README.md
    "API expects: POST /jobs with JobRequest body"

Step 2: AI generates Rust route
  Prompt to Ollama:
    "Generate Rust Actix-web route that:
     - Listens for button click from frontend
     - Creates POST request to /jobs endpoint
     - Returns success/error to frontend
     
     Context:
     - Button emits onClick (no data)
     - API expects JobRequest { shipper_id, pickup, delivery, rate }
     
     Generate complete Rust handler function."

Step 3: AI returns code
  ```rust
  use actix_web::{web, HttpResponse, Result};
  use serde::{Deserialize, Serialize};
  
  #[derive(Deserialize)]
  struct ButtonClick {
      // No data from button
  }
  
  #[derive(Serialize)]
  struct JobRequest {
      shipper_id: String,
      pickup: Location,
      delivery: Location,
      rate: i32,
  }
  
  pub async fn handle_button_to_job_api(
      _click: web::Json<ButtonClick>,
      data: web::Data<AppState>,
  ) -> Result<HttpResponse> {
      // TODO: Get job data from form or state
      let job_request = JobRequest {
          shipper_id: "PLACEHOLDER".to_string(),
          pickup: Location::default(),
          delivery: Location::default(),
          rate: 0,
      };
      
      // Call internal API
      let response = data.http_client
          .post("/jobs")
          .json(&job_request)
          .send()
          .await?;
      
      Ok(HttpResponse::Ok().json(response))
  }
  ```

Step 4: Validate route
  - Check if compiles
  - Check types match
  - Check imports exist

Step 5: Save and hot-reload
  - Write to: backend/src/routes/generated/button_to_job_api.rs
  - Hot-reload Rust backend
  - Connection turns green (success) or red (error)

# Generate new component from template

COMPONENT_TYPE=$1  # atom, molecule, organism
COMPONENT_NAME=$2  # button, search_bar, navbar

if [ -z "$COMPONENT_TYPE" ] || [ -z "$COMPONENT_NAME" ]; then
  echo "Usage: ./create_component.sh <type> <name>"
  echo "Example: ./create_component.sh atom button"
  exit 1
fi

COMPONENT_DIR="lib/components/${COMPONENT_TYPE}s/${COMPONENT_NAME}"

mkdir -p "$COMPONENT_DIR"

# Create widget file
cat > "$COMPONENT_DIR/${COMPONENT_NAME}.dart" << 'EOF'
import 'package:flutter/material.dart';

class ${ComponentName} extends StatelessWidget {
  const ${ComponentName}({Key? key}) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    return Container();
  }
}
EOF

# Create README
cat > "$COMPONENT_DIR/README.md" << 'EOF'
# ${ComponentName}

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| ... | ... | ... | ... |

## Usage

```dart
${ComponentName}()
```
EOF

# Create test file
cat > "$COMPONENT_DIR/${COMPONENT_NAME}.test.dart" << 'EOF'
import 'package:flutter_test/flutter_test.dart';
import '${COMPONENT_NAME}.dart';

void main() {
  testWidgets('${ComponentName} test', (WidgetTester tester) async {
    // TODO: Write tests
  });
}
EOF

echo "Component created: $COMPONENT_DIR"
```

Usage:
```bash
./tools/create_component.sh atom button
./tools/create_component.sh molecule search_bar
./tools/create_component.sh organism navbar
```
